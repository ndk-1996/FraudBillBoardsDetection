create table user
(
email_id varchar(500) primary key,
organization_name varchar(500) not null,
address varchar(500),
contact_number bigint,
password varchar(500)
)

create table billboard
(
billboard_id varchar(500) primary key,
area varchar(500) not null,
billboard_size bigint not null,
cost_per_day bigint
)

create table transactions
(
transaction_id varchar(500) primary key,
transaction_date date not null,
billboard_id varchar(500) not null,
email_id varchar(500) not null,
booking_start_date date,
booking_end_date date,
foreign key (billboard_id) references billboard (billboard_id) on delete cascade,
foreign key (email_id) references user (email_id) on delete cascade
)

create table images
(
img_id varchar(500) primary key,
billboard_id varchar(500) not null,
img_upload_date date not null,
img_location varchar(500) not null,
foreign key (billboard_id) references billboard (billboard_id) on delete cascade
)

create table notifications
(
notification_id varchar(500) primary key,
img_id varchar(500) not null,
content varchar(500),
processed_or_not char(1) not null,
ocr_job_finished_or_not char(1) not null,
foreign key (img_id) references images (img_id) on delete cascade
)
